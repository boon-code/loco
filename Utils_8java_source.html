00128   public static LocationResult locateGsmCell(int cellid, int lac)
00129   {
00130     try
00131     {
00132       URL url = new URL("http://www.google.com/glm/mmap");
00133       HttpURLConnection connection = (HttpURLConnection)url.openConnection();
00134       connection.setRequestMethod("POST");
00135       connection.setDoOutput(true);
00136       connection.setDoInput(true);
00137       connection.connect();
00138       OutputStream ostream = connection.getOutputStream();
00139       ostream.write(MAPS_HEADER);
00140       writeIntBigEndian(ostream, cellid);
00141       writeIntBigEndian(ostream, lac);
00142       ostream.write(MAPS_TRAILER);
00143       ostream.flush();
00144       
00145       DataInputStream istream = new DataInputStream(connection.getInputStream());
00146       istream.skipBytes(3);
00147       int error_code = istream.readInt();
00148       
00149       if (error_code == 0)
00150       {
00151         double latitude  = (double)istream.readInt() / 1000000.0;
00152         double longitude = (double)istream.readInt() / 1000000.0;
00153         return new LocationResult(latitude, longitude);
00154       }
00155       else
00156       {
00157         Log.d(TAG, String.format("locateGsmCell: Google returned error-code: %d", error_code));
00158       }
00159     }
00160     catch(IOException ex)
00161     {
00162       Log.e(TAG, "locateGsmCell: Couldn't retrieve position...", ex);
00163     }
00164     return new LocationResult();
00165   }
