00123     
00124     if (data.length != 4)

00126       Log.e(TAG, "readData: wrong format: expect 4 strings...");

00129     
00130     try
00131     {
00132       m_cellid = Integer.parseInt(data[2]);

00135     catch(NumberFormatException ex)
00136     {
00137       Log.e(TAG, "Illegal format of cellid or lac", ex);
00138       return false;
00139     }
00140     
00141     StringBuilder b = new StringBuilder();
00142     
00143     b.append("mcc: ");
00144     b.append(data[0]);
00145     b.append("\nmnc: ");
00146     b.append(data[1]);

00149     b.append("\nlac: ");

00152     m_txt_data.setText(b.toString());
